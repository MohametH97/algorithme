
## Question 1 — Procedure dot_product

```
PROCEDURE dot_product(v1 : ARRAY OF REAL,
                      v2 : ARRAY OF REAL,
                      n  : INTEGER,
                      VAR ps : REAL)
/* Pass by value    : v1, v2, n               */
/* Pass by reference (VAR) : ps               */
VAR
    i : INTEGER
BEGIN
    ps := 0
    FOR i := 0 TO n-1 DO
        ps := ps + v1[i] * v2[i]
    END_FOR
END_PROCEDURE
```

## Question 2 — Algorithm using the Procedure

```
ALGORITHM VectorOrthogonality
VAR
    v1, v2      : ARRAY[10] OF REAL    /* vectors of max size 10  */
    n           : INTEGER              /* dimension of vectors    */
    nbPairs     : INTEGER              /* number of pairs         */
    ps          : REAL                 /* dot product result      */
    i, k        : INTEGER              /* loop indices            */

BEGIN
    Write("Enter the number of vector pairs : ")
    Read(nbPairs)

    Write("Enter the dimension of the vectors : ")
    Read(n)

    /* Process each pair */
    FOR k := 1 TO nbPairs DO

        Write("--- Pair ", k, " ---")

        /* Input v1 */
        Write("Enter the components of v1 : ")
        FOR i := 0 TO n-1 DO
            Read(v1[i])
        END_FOR

        /* Input v2 */
        Write("Enter the components of v2 : ")
        FOR i := 0 TO n-1 DO
            Read(v2[i])
        END_FOR

        /* Call the procedure (ps passed by reference) */
        CALL dot_product(v1, v2, n, ps)

        /* Orthogonality test */
        IF (ps = 0) THEN
            Write("The vectors are ORTHOGONAL (ps = 0)")
        ELSE
            Write("The vectors are NOT orthogonal (ps = ", ps, ")")
        END_IF

    END_FOR
END
```

## Question 3 — Version using a FUNCTION

```
/* Replace the procedure with a function */

FUNCTION dot_product(v1 : ARRAY OF REAL,
                     v2 : ARRAY OF REAL,
                     n  : INTEGER) : REAL
/* Directly returns the dot product */
VAR
    i  : INTEGER
    ps : REAL
BEGIN
    ps := 0
    FOR i := 0 TO n-1 DO
        ps := ps + v1[i] * v2[i]
    END_FOR
    RETURN ps
END_FUNCTION

/* ------------------------------------------------ */

ALGORITHM VectorOrthogonality_v2
VAR
    v1, v2      : ARRAY[10] OF REAL
    n           : INTEGER
    nbPairs     : INTEGER
    ps          : REAL
    i, k        : INTEGER

BEGIN
    Write("Enter the number of vector pairs : ")
    Read(nbPairs)

    Write("Enter the dimension of the vectors : ")
    Read(n)

    FOR k := 1 TO nbPairs DO

        Write("--- Pair ", k, " ---")

        FOR i := 0 TO n-1 DO
            Read(v1[i])
        END_FOR

        FOR i := 0 TO n-1 DO
            Read(v2[i])
        END_FOR

        /* Call the FUNCTION (result directly assigned) */
        ps := dot_product(v1, v2, n)

        IF (ps = 0) THEN
            Write("The vectors are ORTHOGONAL")
        ELSE
            Write("Dot product = ", ps, " → NOT orthogonal")
        END_IF

    END_FOR
END
